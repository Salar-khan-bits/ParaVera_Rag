Project Overview
ParaVerRAG is a Parallel Verification-Augmented Retrieval system that answers factual questions by:
Retrieving relevant documents in parallel across CPU workers
Scoring retrieved chunks via an OpenCL GPU kernel
Generating an answer using a local LLM (LLaMA 3.2:3B via Ollama)
Verifying the answer with three parallel NLI-based agents (consistency, contradiction, hallucination)
Self-correcting via a feedback loop (up to 3 retries) if verification fails
Hardware target: Consumer laptop, WSL2, integrated GPU via OpenCL  
LLM: LLaMA 3.2:3B served via Ollama at `http://localhost:11434`  
Encoder: `all-MiniLM-L6-v2` (sentence-transformers)  
Verification model: `cross-encoder/nli-deberta-v3-small`

```python
# retrieval/worker.py
# Runs inside a spawned process. Searches one shard of the corpus.

import numpy as np
from retrieval.encoder import encode_batch

def worker_search(args):
    worker_id, shard_docs, query_embedding, top_k = args

    if not shard_docs:
        return []

    texts = [d["text"] for d in shard_docs]
    doc_embeddings = encode_batch(texts)

    # Cosine similarity (embeddings are already normalized)
    scores = doc_embeddings @ query_embedding

    top_indices = np.argsort(scores)[::-1][:top_k]

    results = []
    for idx in top_indices:
        results.append({
            "worker_id": worker_id,
            "doc_id": shard_docs[idx].get("id", f"w{worker_id}:d{idx}"),
            "text": shard_docs[idx]["text"],
            "retrieval_score": float(scores[idx]),
            "scoring_backend": "cpu"
        })

    return results
```

retrieval/parallel_retriever.py
```python
# retrieval/parallel_retriever.py

import json
import os
import numpy as np
from multiprocessing import Pool
from retrieval.encoder import encode
from retrieval.worker import worker_search
from scoring.gpu_scorer import rescore_with_gpu
from config import CORPUS_DIR, NUM_WORKERS, TOP_K_PER_WORKER, FINAL_TOP_K

def load_corpus() -> list[dict]:
    docs = []
    if not os.path.exists(CORPUS_DIR):
        return docs
    for fname in os.listdir(CORPUS_DIR):
        fpath = os.path.join(CORPUS_DIR, fname)
        if fname.endswith(".jsonl"):
            with open(fpath) as f:
                for line in f:
                    docs.append(json.loads(line))
        elif fname.endswith(".json"):
            with open(fpath) as f:
                docs.extend(json.load(f))
    return docs

def split_into_shards(docs: list, n: int) -> list[list]:
    size = max(1, len(docs) // n)
    return [docs[i * size:(i + 1) * size] for i in range(n)]

def retrieve(query: str, corpus_docs: list[dict] = None) -> list[dict]:
    if corpus_docs is None:
        corpus_docs = load_corpus()

    query_embedding = encode(query)
    shards = split_into_shards(corpus_docs, NUM_WORKERS)

    worker_args = [
        (i + 1, shards[i], query_embedding, TOP_K_PER_WORKER)
        for i in range(len(shards))
    ]

    with Pool(processes=NUM_WORKERS) as pool:
        shard_results = pool.map(worker_search, worker_args)

    # Merge all worker results
    all_docs = [doc for shard in shard_results for doc in shard]

    # Re-score with GPU (OpenCL)
    all_docs = rescore_with_gpu(query_embedding, all_docs)

    # Sort by final score and return top-K
    all_docs.sort(key=lambda x: x["score"], reverse=True)
    return all_docs[:FINAL_TOP_K]
```

scoring/kernels/cosine_similarity.cl
```c
/* scoring/kernels/cosine_similarity.cl */

__kernel void cosine_similarity(
    __global const float* query,
    __global const float* docs,
    __global float* scores,
    const int dim,
    const int n_docs
) {
    int doc_idx = get_global_id(0);
    if (doc_idx >= n_docs) return;

    float dot = 0.0f;
    float q_norm = 0.0f;
    float d_norm = 0.0f;

    for (int i = 0; i < dim; i++) {
        float q = query[i];
        float d = docs[doc_idx * dim + i];
        dot   += q * d;
        q_norm += q * q;
        d_norm += d * d;
    }

    float denom = sqrt(q_norm) * sqrt(d_norm);
    scores[doc_idx] = (denom > 0.0f) ? (dot / denom) : 0.0f;
}
```

scoring/gpu_scorer.py
```python
# scoring/gpu_scorer.py

import numpy as np
import os
from config import USE_OPENCL, OPENCL_BONUS, EMBEDDING_DIM
from retrieval.encoder import encode_batch

def rescore_with_gpu(query_embedding: np.ndarray, docs: list[dict]) -> list[dict]:
    if not docs:
        return docs

    if USE_OPENCL:
        try:
            return _opencl_score(query_embedding, docs)
        except Exception as e:
            print(f"[GPU Scorer] OpenCL failed ({e}), falling back to CPU numpy.")

    return _cpu_score(query_embedding, docs)


def _cpu_score(query_embedding, docs):
    texts = [d["text"] for d in docs]
    embeddings = encode_batch(texts)
    scores = embeddings @ query_embedding
    for i, doc in enumerate(docs):
        doc["score"] = float(scores[i])
        doc["scoring_backend"] = "cpu"
    return docs


def _opencl_score(query_embedding, docs):
    import pyopencl as cl

    texts = [d["text"] for d in docs]
    embeddings = encode_batch(texts).astype(np.float32)
    query = query_embedding.astype(np.float32)
    n_docs = len(docs)

    platforms = cl.get_platforms()
    ctx = cl.Context(platforms[0].get_devices())
    queue = cl.CommandQueue(ctx)

    kernel_path = os.path.join(
        os.path.dirname(__file__), "kernels", "cosine_similarity.cl"
    )
    with open(kernel_path) as f:
        kernel_src = f.read()

    program = cl.Program(ctx, kernel_src).build()
    mf = cl.mem_flags

    query_buf = cl.Buffer(ctx, mf.READ_ONLY | mf.COPY_HOST_PTR, hostbuf=query)
    docs_buf  = cl.Buffer(ctx, mf.READ_ONLY | mf.COPY_HOST_PTR, hostbuf=embeddings.flatten())
    scores_buf = cl.Buffer(ctx, mf.WRITE_ONLY, size=n_docs * 4)

    program.cosine_similarity(
        queue, (n_docs,), None,
        query_buf, docs_buf, scores_buf,
        np.int32(EMBEDDING_DIM), np.int32(n_docs)
    )

    gpu_scores = np.empty(n_docs, dtype=np.float32)
    cl.enqueue_copy(queue, gpu_scores, scores_buf).wait()

    for i, doc in enumerate(docs):
        doc["score"] = float(gpu_scores[i]) + OPENCL_BONUS
        doc["scoring_backend"] = "opencl"

    return docs
```

```python
# generation/llm_generator.py

import requests
from config import OLLAMA_URL, LLM_MODEL, LLM_TEMPERATURE, LLM_MAX_TOKENS, LLM_TIMEOUT

def generate_answer(query: str, docs: list[dict], feedback: str = None) -> str:
    context = "\n".join([
        f"Document {i+1}: {d['text']}"
        for i, d in enumerate(docs)
    ])

    feedback_block = ""
    if feedback:
        feedback_block = f"""
A previous answer was rejected. Reason:
{feedback}

Fix this in your new answer. Be more conservative — only state what the documents explicitly say.
"""

    prompt = f"""You are a factual question answering assistant.
Answer the question using ONLY the information in the documents below.
Be concise and direct. Do not add facts not present in the documents.
{feedback_block}
Documents:
{context}

Question: {query}

Answer:"""

    try:
        response = requests.post(
            OLLAMA_URL,
            json={
                "model": LLM_MODEL,
                "prompt": prompt,
                "stream": False,
                "options": {
                    "temperature": LLM_TEMPERATURE,
                    "num_predict": LLM_MAX_TOKENS
                }
            },
            timeout=LLM_TIMEOUT
        )
        result = response.json()
        return result["response"].strip()

    except requests.exceptions.ConnectionError:
        raise RuntimeError(
            "Ollama is not running. Start it with: ollama serve"
        )
    except Exception as e:
        raise RuntimeError(f"LLM generation failed: {e}")
```

```python
# verification/consistency_agent.py
# Checks: does the answer match what the retrieved documents say?

from transformers import pipeline
from config import NLI_MODEL, CONSISTENCY_THRESHOLD

_nli = None

def get_nli():
    global _nli
    if _nli is None:
        _nli = pipeline("text-classification", model=NLI_MODEL)
    return _nli

def check_consistency(answer: str, docs: list[dict]) -> dict:
    if not docs:
        return {"agent": "consistency", "passed": False,
                "score": 0.0, "feedback": "No documents retrieved."}

    nli = get_nli()
    best_score = 0.0

    for doc in docs:
        text = doc["text"] if isinstance(doc, dict) else doc
        result = nli(f"{text} [SEP] {answer}")[0]
        label = result["label"].upper()
        raw = result["score"]

        if label == "ENTAILMENT":
            score = raw
        elif label == "NEUTRAL":
            score = raw * 0.3
        else:
            score = 0.0

        best_score = max(best_score, score)

    passed = best_score >= CONSISTENCY_THRESHOLD
    return {
        "agent": "consistency",
        "passed": passed,
        "score": round(best_score, 4),
        "feedback": "" if passed else "Answer is inconsistent with retrieved documents."
    }
```

```python
# Checks: does the answer directly contradict any retrieved document?

from transformers import pipeline
from config import NLI_MODEL, CONTRADICTION_THRESHOLD

_nli = None

def get_nli():
    global _nli
    if _nli is None:
        _nli = pipeline("text-classification", model=NLI_MODEL)
    return _nli

def check_contradiction(answer: str, docs: list[dict]) -> dict:
    if not docs:
        return {"agent": "contradiction", "passed": True,
                "score": 1.0, "feedback": ""}

    nli = get_nli()
    max_contradiction = 0.0

    for doc in docs:
        text = doc["text"] if isinstance(doc, dict) else doc
        result = nli(f"{text} [SEP] {answer}")[0]
        label = result["label"].upper()
        raw = result["score"]

        if label == "CONTRADICTION":
            max_contradiction = max(max_contradiction, raw)

    passed = max_contradiction < CONTRADICTION_THRESHOLD
    return {
        "agent": "contradiction",
        "passed": passed,
        "score": round(1.0 - max_contradiction, 4),
        "feedback": "" if passed else "Answer contradicts retrieved documents."
    }
```

```python
# verification/hallucination_agent.py
# Checks: does the answer contain facts not grounded in any retrieved document?

from transformers import pipeline
from config import NLI_MODEL, HALLUCINATION_THRESHOLD

_nli = None

def get_nli():
    global _nli
    if _nli is None:
        _nli = pipeline("text-classification", model=NLI_MODEL)
    return _nli

def check_hallucination(answer: str, docs: list[dict]) -> dict:
    if not docs:
        return {"agent": "hallucination", "passed": False,
                "score": 0.0, "feedback": "No evidence documents provided."}

    nli = get_nli()
    best_score = 0.0

    for doc in docs:
        text = doc["text"] if isinstance(doc, dict) else doc
        result = nli(f"{text} [SEP] {answer}")[0]
        label = result["label"].upper()
        raw = result["score"]

        if label == "ENTAILMENT":
            score = raw
        elif label == "NEUTRAL":
            score = raw * 0.4
        else:
            score = 0.0

        # KEY FIX: take MAX across all docs, not average
        # If any single doc supports the answer, it is not hallucinated
        best_score = max(best_score, score)

    passed = best_score >= HALLUCINATION_THRESHOLD
    return {
        "agent": "hallucination",
        "passed": passed,
        "score": round(best_score, 4),
        "feedback": "" if passed else "Reduce claims not explicitly present in evidence."
    }
```

```python
# verification/agent_runner.py
# Runs all 3 verification agents concurrently on separate threads.

from concurrent.futures import ThreadPoolExecutor, as_completed
from verification.consistency_agent import check_consistency
from verification.contradiction_agent import check_contradiction
from verification.hallucination_agent import check_hallucination

def run_agents(answer: str, docs: list[dict]) -> dict:
    agents = {
        "consistency":   (check_consistency,   answer, docs),
        "contradiction": (check_contradiction,  answer, docs),
        "hallucination": (check_hallucination,  answer, docs),
    }

    results = {}

    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = {
            executor.submit(fn, ans, dc): name
            for name, (fn, ans, dc) in agents.items()
        }
        for future in as_completed(futures):
            name = futures[future]
            results[name] = future.result()

    all_passed = all(r["passed"] for r in results.values())

    # Collect feedback from failed agents
    feedback_parts = [
        r["feedback"] for r in results.values()
        if not r["passed"] and r["feedback"]
    ]
    combined_feedback = " ".join(feedback_parts)

    return {
        "all_passed": all_passed,
        "results": list(results.values()),
        "feedback": combined_feedback
    }
```

correction/self_corrector.py
```python
# correction/self_corrector.py
# Retry loop: if verification fails, regenerate with agent feedback (max 3 times).

from generation.llm_generator import generate_answer
from verification.agent_runner import run_agents
from config import MAX_CORRECTION_ATTEMPTS

def correct(query: str, docs: list[dict], initial_answer: str,
            initial_verification: dict) -> tuple[str, int, dict]:

    answer = initial_answer
    verification = initial_verification

    for attempt in range(1, MAX_CORRECTION_ATTEMPTS + 1):
        if verification["all_passed"]:
            return answer, attempt, verification

        feedback = verification.get("feedback", "")
        answer = generate_answer(query, docs, feedback=feedback)
        verification = run_agents(answer, docs)

    # Return best available answer after max attempts
    return answer, MAX_CORRECTION_ATTEMPTS, verification
```

main.py
```python
# main.py
# Entry point. Run as: python main.py "Your question here"

import sys
import json
import time
from retrieval.parallel_retriever import retrieve, load_corpus
from generation.llm_generator import generate_answer
from verification.agent_runner import run_agents
from correction.self_corrector import correct

def run_pipeline(query: str) -> dict:
    start = time.time()

    # Stage 1: Load corpus
    corpus_docs = load_corpus()

    # Stage 2: Parallel retrieval + GPU re-scoring
    retrieved_docs = retrieve(query, corpus_docs)

    # Stage 3: Generate initial answer
    initial_answer = generate_answer(query, retrieved_docs)

    # Stage 4: Verify with 3 parallel agents
    verification = run_agents(initial_answer, retrieved_docs)

    # Stage 5: Self-correct if needed
    final_answer, attempts, final_verification = correct(
        query, retrieved_docs, initial_answer, verification
    )

    latency = time.time() - start

    return {
        "query": query,
        "answer": final_answer,
        "verified": final_verification["all_passed"],
        "attempts": attempts,
        "verification": final_verification,
        "retrieved_docs": retrieved_docs,
        "latency_seconds": round(latency, 5)
    }


if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python main.py \"Your question here\"")
        sys.exit(1)

    query = sys.argv[1]
    result = run_pipeline(query)
    print(json.dumps(result, indent=2))
```

evaluation/metrics.py
```python
# evaluation/metrics.py

import re
import string
from collections import Counter

def normalize(text: str) -> str:
    text = text.lower()
    text = re.sub(r'\b(a|an|the)\b', ' ', text)
    text = ''.join(c for c in text if c not in string.punctuation)
    return ' '.join(text.split())

def exact_match(prediction: str, ground_truth: str) -> int:
    return int(normalize(prediction) == normalize(ground_truth))

def f1_score(prediction: str, ground_truth: str) -> float:
    pred_tokens  = normalize(prediction).split()
    truth_tokens = normalize(ground_truth).split()

    if not pred_tokens or not truth_tokens:
        return 0.0

    common = Counter(pred_tokens) & Counter(truth_tokens)
    num_common = sum(common.values())

    if num_common == 0:
        return 0.0

    precision = num_common / len(pred_tokens)
    recall    = num_common / len(truth_tokens)
    return 2 * precision * recall / (precision + recall)

def compute_metrics(predictions: list[dict]) -> dict:
    em_scores = [exact_match(p["answer"], p["gold"]) for p in predictions]
    f1_scores = [f1_score(p["answer"],   p["gold"]) for p in predictions]
    latencies = [p.get("latency", 0)                for p in predictions]

    return {
        "exact_match":    round(sum(em_scores) / len(em_scores), 4),
        "f1":             round(sum(f1_scores) / len(f1_scores), 4),
        "avg_latency_s":  round(sum(latencies) / len(latencies), 4),
        "total_questions": len(predictions)
    }
```

```python
# evaluation/run_eval.py
# Run: python evaluation/run_eval.py --dataset nq --limit 100

import argparse
import json
import os
from main import run_pipeline
from evaluation.metrics import compute_metrics

def load_nq(path: str, limit: int) -> list[dict]:
    data = []
    with open(path) as f:
        for i, line in enumerate(f):
            if i >= limit:
                break
            item = json.loads(line)
            data.append({
                "question": item["question"],
                "gold": item["annotations"][0]["short_answers"][0]["text"]
                        if item.get("annotations") else item.get("answer", "")
            })
    return data

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--dataset", choices=["nq", "triviaqa"], default="nq")
    parser.add_argument("--limit", type=int, default=100)
    args = parser.parse_args()

    dataset_path = os.path.join("data", args.dataset, "test.jsonl")
    samples = load_nq(dataset_path, args.limit)

    predictions = []
    for i, sample in enumerate(samples):
        print(f"[{i+1}/{len(samples)}] {sample['question']}")
        result = run_pipeline(sample["question"])
        predictions.append({
            "question": sample["question"],
            "gold":     sample["gold"],
            "answer":   result["answer"],
            "verified": result["verified"],
            "latency":  result["latency_seconds"]
        })

    metrics = compute_metrics(predictions)
    print("\n=== Evaluation Results ===")
    print(json.dumps(metrics, indent=2))

    with open("logs/eval_results.json", "w") as f:
        json.dump({"metrics": metrics, "predictions": predictions}, f, indent=2)

if __name__ == "__main__":
    main()
```

Known Issues & Fixes
Issue	Cause	Fix
`verified: false` on correct answers	Hallucination agent using mean score across all docs	Use `max()` not `mean()` — fixed in hallucination_agent.py above
`"Draft answer for..."` in output	llm_generator.py using placeholder	Replaced with real Ollama HTTP call
`latency: 0.03s` (impossibly fast)	LLM not being called	Confirm Ollama is running before main.py
OpenCL not found	iGPU not visible in WSL	Run `clinfo`; if empty, gpu_scorer falls back to CPU automatically
NLI model slow on first call	Model downloads on first use	Pre-warm with a dummy call at startup
